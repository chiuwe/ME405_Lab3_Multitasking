00062 void task_source::run (void)

00068     // This is the task loop. Once the task has been initialized in the code just
00069     // above, the task loop runs, and it keeps running until the power is shut off
00070     for (;;)
00071     {
00072         // Create some data and send it to the sink task in the queue
00073         some_data = random ();
00074         p_queue_1->put (some_data & 0xFF00FF00);
00075 
00076         // Write standard values here; we'll check what is received by the sink
00077         if (toggler)
00078         {
00079             p_share_1->put (0x12345678);
00080             *p_glob_of_probs = 0x12345678;
00081         }
00082         else
00083         {
00084             p_share_1->put (0x87654321);
00085             *p_glob_of_probs = 0x87654321;
00086         }
00087         toggler ^= 0x01;
00088 
00089         // Delay a while to give other tasks time to run. We don't want to delay too
00090         // long, as doing so will reduce the probability of causing the data errors
00091         // which this task is supposed to be causing
00092         delay (1);
00093 
00094         // Increment the counter so we can check how many times this task has run
00095         runs++;
