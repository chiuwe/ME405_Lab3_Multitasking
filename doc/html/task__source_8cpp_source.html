62 void task_source::run (void)
63 {
64  // Set up by declaring and initializing variables
65  uint32_t some_data = 0;
66  uint8_t toggler = false;
67 
68  // This is the task loop. Once the task has been initialized in the code just
69  // above, the task loop runs, and it keeps running until the power is shut off
70  for (;;)
71  {
72  // Create some data and send it to the sink task in the queue
73  some_data = random ();
74  p_queue_1->put (some_data & 0xFF00FF00);
75 
76  // Write standard values here; we'll check what is received by the sink
77  if (toggler)
78  {
79  p_share_1->put (0x12345678);
80  *p_glob_of_probs = 0x12345678;
81  }
82  else
83  {
84  p_share_1->put (0x87654321);
85  *p_glob_of_probs = 0x87654321;
86  }
87  toggler ^= 0x01;
88 
89  // Delay a while to give other tasks time to run. We don't want to delay too
90  // long, as doing so will reduce the probability of causing the data errors
91  // which this task is supposed to be causing
92  delay (1);
93 
94  // Increment the counter so we can check how many times this task has run
95  runs++;
96  }
97 }
