75 void task_user::run (void)
76 {
77  char char_in; // Character read from serial device
78  time_stamp a_time; // Holds the time so it can be displayed
79 
80  // Right before running the loop, print the help message so the user knows which
81  // keys to press to do something
82  print_help_message ();
83 
84  // This is the task loop. Once the task has been initialized in the code just
85  // above, the task loop runs, and it keeps running until the power is shut off.
86  // In the loop, we check for characters typed into the serial port and also check
87  // for characters in the print queue; those characters are to be displayed
88  for (;;)
89  {
90  // Check for any characters which might have been typed by the user
91  if (p_serial->check_for_char ())
92  {
93  char_in = p_serial->getchar (); // Read the character
94 
95  switch (char_in)
96  {
97  // The 't' command asks what time it is now
98  case 'n':
99  *p_serial << (a_time.set_to_now ()) << endl;
100  break;
101 
102  // The 's' command means dump all the tasks' stacks for examination
103  case 's':
104  print_task_stacks (p_serial);
105  break;
106 
107  // Pressing 'v' gives the version number and setup of this program
108  case 'v':
109  show_status ();
110  break;
111 
112  // A '?' or 'h' is a plea for help; respond with a help message
113  case '?':
114  case 'h':
115  print_help_message ();
116  break;
117 
118  // The number 3 is control-C, which means reset the AVR processor
119  case (3):
120  *p_serial << PMS ("Resetting AVR") << endl;
121  wdt_enable (WDTO_120MS);
122  for (;;)
123  {
124  }
125  break;
126 
127  // If the character isn't recognized, ask: What's That Function?
128  default:
129  p_serial->putchar (char_in);
130  *p_serial << PMS (":WTF?") << endl;
131  break;
132  };
133  }
134  // If no character was typed in, check the print queue to see if another task
135  // has sent this task something to be printed
136  else if (print_ser_queue->check_for_char ())
137  {
138  p_serial->putchar (print_ser_queue->getchar ());
139  }
140  // If no character has been typed and no other task sent us something to print,
141  // wait for approximately a millisecond before we check again
142  else
143  {
144  vTaskDelay (configTICK_RATE_HZ / 1000);
145  }
146 
147  // We've made it safely through the loop one more time; claim some credit
148  runs++;
149  }
150 }
