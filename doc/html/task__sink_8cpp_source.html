79 void task_sink::run (void)
80 {
81  // Setup
82  uint32_t received_data = 0x12345678;
83 
84  // Wait a moment before starting the loop so that the source task has time to make
85  // some data available
86  delay (10);
87 
88  // This is the task loop. Once the task has been initialized in the code just
89  // above, the task loop runs, and it keeps running until the power is shut off
90  for (;;)
91  {
92  // Check the queue from the source task. If there's data, read it; if not, go
93  // and check the shared data
94  if (p_queue_1->not_empty ())
95  {
96  received_data = p_queue_1->get ();
97  if ((received_data & 0x00FF00FF) != 0)
98  {
99  queue_errors++;
100  *print_ser_queue << PMS ("ERROR in queue, got ") << hex
101  << received_data << dec << endl;
102  }
103  }
104 
105  // Check the data sent by other means as well
106  received_data = p_share_1->get ();
107  if (received_data != 0x12345678 && received_data != 0x87654321)
108  {
109  share_errors++;
110  *print_ser_queue << PMS ("ERROR in share, got ") << hex << received_data
111  << dec << endl;
112  }
113  received_data = *p_glob_of_probs;
114  if (received_data != 0x87654321 && received_data != 0x12345678)
115  {
116  global_errors++;
117 
118  // Since this is a high priority task, we need to delay for a few ticks
119  // each time we get an error, or the user interface task never gets to run
120  delay (20);
121 
122  // We can't print this out every time because if we do, there are so many
123  // errors that one can hardly see anything else on the screen
124 // *print_ser_queue << PMS ("ERROR in global, got ") << hex << received_data
125 // << dec << endl;
126  }
127  runs++;
128  }
129 }
